git init - для инициализации репозитория(хранилища данных)
git add .- для добавления всех изменений в следующий коммит
git commit -m " "  -создание коммита с комментарием
git checkout -b ...  - создание новой ветки с переключением на неё(ветка клонируентся с текущей)
git checkout ...  - переключение уже на существующую ветку
git merge -  команда для слияния веток
